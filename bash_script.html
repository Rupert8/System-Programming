Скрипт авторизації

        #!/bin/bash

        # Визначаємо ім'я користувача та пароль
        CORRECT_USERNAME="admin"
        CORRECT_PASSWORD="password123"
        
        # Запитуємо у користувача ім'я користувача
        read -p "Введіть ім'я користувача: " USERNAME
        
        # Запитуємо у користувача пароль (приховане введення)
        read -sp "Введіть пароль: " PASSWORD
        echo # Для переведення на новий рядок після прихованого вводу
        
        # Перевіряємо введені дані
        if [[ "$USERNAME" == "$CORRECT_USERNAME" && "$PASSWORD" == "$CORRECT_PASSWORD" ]]; then
            echo "Авторизація успішна! Ласкаво просимо, $USERNAME."
        else
            echo "Неправильне ім'я користувача або пароль. Спробуйте ще раз."
        fi
